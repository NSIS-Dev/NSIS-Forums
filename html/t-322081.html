!include"logiclib.nsh"

***91;...***93;

Section
${Select}$LANGUAGE
${Case}${LANG_ENGLISH}
MessageBoxMB_OK"Englishfiles"
${Case}${LANG_GERMAN}
MessageBoxMB_OK"Germanfiles"
${Case}${LANG_FRENCH}
MessageBoxMB_OK"Frenchfiles"
${EndSelect}
SectionEnd
