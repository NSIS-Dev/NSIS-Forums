Animaether
30th August 2010 18:45 UTC

!include "logiclib.nsh"

***91;...***93;

Section
    ${Select} $LANGUAGE
    ${Case} ${LANG_ENGLISH}
        MessageBox MB_OK "English files"
    ${Case} ${LANG_GERMAN}
        MessageBox MB_OK "German files"
    ${Case} ${LANG_FRENCH}
        MessageBox MB_OK "French files"
    ${EndSelect}
SectionEnd
