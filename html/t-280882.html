  loop:
    EnumRegKey $1 HKLM Software $2
    StrCmp $1 "v2.0.50727" done
    IntOp $2 $2 + 1
    !StrCmp $1 "" loop

${Do}
        ClearErrors
        EnumRegKey $1 HKLM SOFTWARE\Microsoft\.NETFramework $2
        ${If} ${Errors}
                ${Break}
        ${EndIf}

        ${If} $1 == "v2.0.50727"
                MessageBox MB_OK found
                ${Break}
        ${EndIf}

        IntOp $2 $2 + 1
${LoopWhile} $1 != ""
