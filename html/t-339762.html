Function lic_show
StrCmp $R1 '0' btncheck

FindWindow $1 "#32770" "" $HWNDPARENT

ShowWindow $2 ${SW_HIDE}

GetDlgItem $1 $1 1000
System::Call "User32::SetWindowPos(i, i, i, i, i, i, i) b ($1, 0, 0, 40, 0, 0, 1)"
btncheck:
strcmp '$0' '2' 0 end
GetDlgItem $1 $HWNDPARENT 3
EnableWindow $1 0
end:
