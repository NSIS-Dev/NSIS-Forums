finaly:

Section "Permissions"

System::Call kernel32::GetComputerName(t.r0,*i${NSIS_MAX_STRLEN})
StrCpy $0 IUSR_$0
DetailPrint $0

AccessControl::GrantOnFile "$INSTDIR/yourFileName" $0 "FullAccess"
pop $0
DetailPrint "-IISAnonymousUser - $0 - DONE"


SectionEnd
