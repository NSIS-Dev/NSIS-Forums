Function GetWinampInstPath

Push $0
Push $1
Push $2
ReadRegStr $0 HKLM \

StrCpy $1 $0 1 0

StrCpy $0 $0 "" 1

rqloop:
StrCpy $2 $0 1 $1
StrCmp $2 '"' rqdone
StrCmp $2 "" rqdone
IntOp $1 $1 + 1
Goto rqloop
rqdone:

getparent:

StrCpy $1 -1
gploop:
StrCpy $2 $0 1 $1
StrCmp $2 "" gpexit
StrCmp $2 "\" gpexit

gpexit:
StrCpy $0 $0 $1

StrCmp $0 "" fin
IfFileExists $0\winamp.exe fin
