system::call *(&i16,l)i.s

system::call 'kernel32::GetLocalTime(isr0)'

>IntOp $1 $0 + 16

system::call 'kernel32::SystemTimeToFileTime(ir0,ir1)'

>system::call *$1(l.r1)

>system::free $0

system::Int64Op $1 / 10000000

Pop$1

system::Int64Op $1 - 11644473600

Pop$1

DetailPrint$1
