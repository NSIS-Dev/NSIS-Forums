Try

# Installer languages
!insertmacro MUI_LANGUAGE English
!insertmacro MUI_LANGUAGE French

Section $(SEC0000_NAME) SEC0000
    File abc.txt
SectionEnd

LangString SEC0000_NAME     ${LANG_ENGLISH} "Main Program"
LangString SEC0000_NAME     ${LANG_FRENCH}  "Programme"
