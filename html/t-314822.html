You can try this too:


GetVersion::WindowsPlatformArchitecture
Pop $R1

${If} $R1 == "32"
   MessageBox MB_OK "32 bits"
${ElseIf} $R1 == "64"
   MessageBox MB_OK "64 bits"
${Endif}
