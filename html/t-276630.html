Use GetACP.

System::Call kernel32::GetACP()i.r0
${If} $0 == somecodepage
StrCpy $LANGUAGE ${LANG_SOMETHING}
${ElseIf} $0 == someothercodepage
StrCpy $LANGUAGE ${LANG_SOMETHINGELSE}
${Else}
StrCpy $LANGUAGE ${LANG_DEFAULT}
${EndIf}
