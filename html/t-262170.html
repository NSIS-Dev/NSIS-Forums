outfile test.exe
licensedata '${NSISDIR}\license.txt'

LoadLanguageFile "${NSISDIR}\Contrib\Language files\French.nlf"

Insttype french
Insttype greek

page license
page components
page instfiles

section /o "english" sec1
   sectionin 1

sectionend

section /o "french" sec2
   sectionin 2

sectionend

section /o "greek" sec3
   sectionin 3

sectionend

function .onInit
        Push ""
        Push ${LANG_ENGLISH}
        Push English
        Push ${LANG_FRENCH}
        Push French
        Push ${LANG_GREEK}
        Push Greek
        Push A
        LangDLL::LangDialog "Installer Language" "Please select the language of the installer."
        Pop $LANGUAGE
        StrCmp $LANGUAGE "cancel" 0 +2
        Abort

        StrCmp $LANGUAGE '1033' 0 +2
        SectionSetFlags ${sec1} 1

        StrCmp $LANGUAGE '1036' 0 +2
        SectionSetFlags ${sec2} 1

        StrCmp $LANGUAGE '1032' 0 +2
        SectionSetFlags ${sec3} 1
functionend
