!include LogicLib.nsh

Function .onVerifyInstDir
        FindWindow $R0 "#32770" "" $HWNDPARENT
        GetDlgItem $R0 $R0 1019
        ${If} ${FileExists} "$INSTDIR\file.ext"
                SetCtlColors $R0 000000 00FF00
        ${Else}
                SetCtlColors $R0 000000 FF0000
        ${EndIf}
FunctionEnd
