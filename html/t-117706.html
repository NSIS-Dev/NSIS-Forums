Use LangString.

ex:


LangString Title_Sec1 ${LANG_ENGLISH} "Section1"
LangString Title_Sec1 ${LANG_AnotherLanguageName} "Section1 in other languages"

Section $(Title_Sec1)
;some codes
SectionEnd
