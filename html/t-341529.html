${registry::Open} "HKEY_CURRENT_USER\Software\my software" "/K=0 /V=1 /S=0 /B=1 /N='LocalFileName'" $0

StrCmp $4 '' close
${WordFind} "$3" "settings" "E+2*}" $6

goto loop
