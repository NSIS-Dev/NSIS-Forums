You can't use OutFile as a variable.
Do something like

!ifdef PLATFORM64
   !define OutFileName setup64.exe
!else
   !define OutFileName setup32.exe
!endif
OutFile ${OutFileName}
...
${StrContains} $0 "64" ${OutFileName}
StrCmp $0 "" Notfound
...
