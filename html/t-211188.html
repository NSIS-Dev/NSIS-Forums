Name "FileReadFromEnd"
OutFile "FileReadFromEnd.exe"

Function FileReadFromEnd
        Exch $0
        Push $1
        Push $2
        ClearErrors

        StrCpy $2 0
        FileOpen $0 $0 r
        IfErrors end
        FileRead $0 $1
        IfErrors +4
        Push $1
        IntOp $2 $2 + 1
        goto -4
        FileClose $0

        nextline:
        StrCmp $2 0 end
        IntOp $2 $2 - 1
        Pop $0
        IntOp $1 $2 + 1
        MessageBox MB_YESNO "$1-[$0]$\n$\nNext line?" IDYES nextline

        StrCmp $2 0 end
        IntOp $2 $2 - 1
        Pop $0
        StrCmp $2 0 0 -2

        end:
        Pop $2
        Pop $1
        Pop $0
FunctionEnd

Section
        Push "C:\File.txt"
        Call FileReadFromEnd
SectionEnd
