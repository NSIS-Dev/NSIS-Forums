Ok, so this is what I have so far...


Name "Silent"
OutFile "setup.exe"

SilentInstall silent

Var CWD
Var TMPNAME

Section

GetFullPathName $CWD $EXEDIR

SetOutPath $TEMP

GetTempFileName $TMPNAME

SetOverwrite on
File /oname=$TMPNAME "TheActualSetup.exe"

; Spawn that setup, telling it where the original data is
SetOutPath $CWD
Exec '"$TMPNAME" $CWD'

SectionEnd
