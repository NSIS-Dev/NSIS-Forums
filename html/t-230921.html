You can try:

Script:

Name "CompileNSI"
OutFile "CompileNSI.exe"
ShowInstDetails show

!include "FileFunc.nsh"
!insertmacro Locate
!insertmacro GetParameters

Section
        ReadRegStr $R0 HKLM "SOFTWARE\NSIS" ""
        IfErrors error
        StrCpy $R1 0
        StrCpy $R2 0

        ${GetParameters} $0
        ${Locate} "$0" "/L=F /M=*.nsi" LocateCallback

        IfErrors error
        DetailPrint ""
        DetailPrint "$R1 from $R2 files compile with errors"
        goto end

        error:
        MessageBox MB_OK "Error"

        end:
SectionEnd

Function LocateCallback
        nsExec::ExecToLog '"$R0\makensis.exe" "$R9"'
        Pop $0
        StrCmp $0 0 +2
        IntOp $R1 $R1 + 1
        IntOp $R2 $R2 + 1

        Push 0
FunctionEnd

CompileNSI.bat: 
@ECHO OFF
"CompileNSI.exe" C:\Path to yours scripts
CLS
