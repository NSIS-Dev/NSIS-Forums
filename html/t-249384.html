  StrCpy $0 0
  EnumRegKey $1  HKCU "SOFTWARE\Adobe\Acrobat Reader" $0
  StrCmp $1 "" 0 regclean1
    DeleteRegKey HKCU "SOFTWARE\Adobe\Acrobat Reader"
  regclean1:
