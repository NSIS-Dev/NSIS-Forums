or

Name "Output"
OutFile "Output.exe"

!include "WordFunc.nsh"
!insertmacro WordFind

Section
        StrCpy $0 "xy,ab,dc"

        ${WordFind} "$0" "," "+1" $1
        ${WordFind} "$0" "," "+2" $2
        ${WordFind} "$0" "," "+3" $3

        MessageBox MB_OK '$$1={$1}$\n$$2={$2}$\n$$3={$3}'
SectionEnd
