!include "Sections.nsh"

Section "bla" SecName

SectionEnd

Function .onInit

  Push $R0

    SectionGetFlags ${SecName} $R0
    IntOp $R0 $R0 - ${SF_SELECTED}
    SectionSetFlags ${SecName} $R0
    
  Pop $R0
  
FunctionEnd
