ReadIniStr $0 <yourInstallOptions.ini> <DropList Field> <State>
strcmp $0 'Holidays' 0 +3
strcpy $0 '22'
goto next
strcmp $0 'Shopping' 0 +3
strcpy $0 '33'
goto next
..................
..................
next:
