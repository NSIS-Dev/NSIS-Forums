The easy solution is of course to put the version number in "Name", if you don't want to do that you can use this ugly workaround:

PHP Code:

outfiletest.exe
requestexecutionleveluser

Name"Test"
 !defineVERSION1.2.3

 !macroCreateSetupCaptionLangstringlangprefixsuffix
 !tempfiletnsi
 !appendfile"${tnsi}"'outfile"${tnsi}.exe"$\nrequestexecutionleveluser$\nLoadLanguageFile"${NSISDIR}\Contrib\Languagefiles\${lang}.nlf"$\n'
 !appendfile"${tnsi}"'section$\nFileOpen$0"${tnsi}.nsh"w$\n'
 !appendfile"${tnsi}"'FileWrite$0`!defineSetupCaption"$(^SetupCaption)"`$\nsectionend$\nName"xxx"$\n'
 !system'"${NSISDIR}\makensis""${tnsi}"'=0
 !delfile"${tnsi}"
 !system'"${tnsi}.exe"/S'=0
 !delfile"${tnsi}.exe"
 !include"${tnsi}.nsh"
 !delfile"${tnsi}.nsh"
 !undeftnsi
 !searchreplaceSetupCaption"${SetupCaption}""xxx""${prefix}"
 !ifndefLANG_${lang}
 LoadLanguageFile"${NSISDIR}\Contrib\Languagefiles\${lang}.nlf"
 !endif
 LangString"^SetupCaption"${LANG_${lang}}"${SetupCaption}${suffix}"
 !undefSetupCaption
 !macroend
 !macroLOAD_MUILANGUAGE_AND_SETUPCAPTIONlangprefixsuffix
 !insertmacroMUI_LANGUAGE"${lang}"
 !insertmacroCreateSetupCaptionLangstring"${lang}""$(^Name)v${VERSION}"""
 !macroend

 !include"MUI2.nsh"
 !insertmacroMUI_PAGE_INSTFILES

 !insertmacroLOAD_MUILANGUAGE_AND_SETUPCAPTION"English""$(^Name)v${VERSION}"""
 !insertmacroLOAD_MUILANGUAGE_AND_SETUPCAPTION"Swedish""$(^Name)v${VERSION}"""
 !insertmacroLOAD_MUILANGUAGE_AND_SETUPCAPTION"Norwegian""$(^Name)v${VERSION}"""
 !insertmacroMUI_RESERVEFILE_LANGDLL

 Function.onInit
 !insertmacroMUI_LANGDLL_DISPLAY
FunctionEnd

section
sectionend
