Name "Output"
OutFile "Output.exe"

!include LogicLib.nsh

Function .onInit
  StrCpy $R0 "c:\program files\my dir\"
  StrCpy $0 ''
  StrCpy $1 0

 ${Do}
  StrCpy $2 $R0 1 $1
  StrCpy $0 "$0$2"
  IntOp $1 $1 + 1
  ${Select} $2 
   ${Case} "\"
    StrCpy $0 "$0$2"
    StrCpy $2 "$2\"
   ${Case} ""
    messagebox mb_ok "$0"
    ${ExitDo}
   ${CaseElse}
  ${EndSelect}
 ${Loop}
FunctionEnd

Section -

SectionEnd
