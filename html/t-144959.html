Function .onInit
Call GetParameters
Pop $0
StrCmp $0 "/S" end_function
StrCmp $0 "/a" end_function
MessageBox MB_YESNO|MB_ICONQUESTION "You have ran the installer in silent mode.$\r$\nContinue? IDNO +2
Exec "My_Install_Package.exe /S"
Abort
end_function:
FunctionEnd
