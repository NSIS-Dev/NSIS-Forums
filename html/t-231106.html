Name "Output"
OutFile "Output.exe"

!include "WordFunc.nsh"
!insertmacro WordReplace

Section
        ${WordReplace} " xyz abc " " " "" "{}" $R0
        # $R0="xyz abc"

        MessageBox MB_OK '$$0={$0}'
SectionEnd
