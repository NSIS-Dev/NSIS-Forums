!define BTN_NEXT   1
!define BTN_CANCEL 2
!define BTN_BACK   3

Function ChangeButtonText
         GetDlgItem $0 $HWNDPARENT ${BTN_I_WANT_TO_CHANGE}
         SendMessage $0 ${WM_SETTEXT} 0 "STR:text"
FunctionEnd
