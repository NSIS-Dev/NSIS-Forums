Something like:

# ODBC_ADD_DSN = 1
# ODBC_CONFIG_DSN = 2
# ODBC_REMOVE_DSN = 3 

System::Call 'ODBCCP32::SQLConfigDataSource( ,i 1,t "Microsoft Excel Driver (*.xls)",\
                                                  t "DSN=New Excel Data Source\0\
                                                     Description=New Excel Data Source\0\
                                                     FileType=Excel\0\
                                                     DataDirectory=C:\\ExcelData\0\
                                                     MaxScanRows=20\0")i .r0'
