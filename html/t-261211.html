Function .onInit

;Allow only one installer instance
System::Call 'kernel32::CreateMutexA(i 0, i 0, t "5C00D2E5-758B-4F49-A0BB-03E596E0B470") i .r1 ?e'
Pop $R0
StrCmp $R0 0 DoRunInstall
Abort
DoRunInstall:
