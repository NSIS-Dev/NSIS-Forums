For anyone interested, this is my code:
Possibly buggy, particulally in the return parts, but it seems to work...

outFile "installer.exe"
!define writeHex "!insertmacro writeHex"

!macro writeHex File String
Push "${String}"
Push "${File}"
Call writeHex
!macroend

Function writeHex
ClearErrors
Exch $0 ; Stack: $0 <string>
Exch ; Stack: <string> $0
Exch $1 ; Stack: $1 $0
Push $2 ; Stack: $2 $1 $0
Push $3
Push $4

Push 0
Pop $4
FileOpen $2 "$0" "w"

Loop:
StrCpy $3 $1 2 $4 ; Get next characters
StrCmp $3 "" Done
IntFmt $3 "%d" "0x$3"
FileWriteByte $2 $3
IntOp $4 $4 + 2
Goto Loop

Done:
FileClose $2
Pop $2 ; Stack: $1 $0
Pop $1 ; Stack: $0
Pop $0 ; Stack: -empty-

FunctionEnd

Section Test
${writeHex} "c:\temp\test\a_log_file.txt" "65666768"
SectionEnd
