!include WinVer.nsh

>${If} ${AtLeastWinXP}

   System::Alloc 36

    pop$0

   ${If} $0 <> 0

        System::Call 'kernel32::GetNativeSystemInfo(i $0)'

       System::Call "*$0(&i2.r1)"

       MessageBox mb_ok wProcessorArchitecture=$1

        System::Free $0

    ${EndIf}

${EndIf}
