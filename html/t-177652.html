Function OPENIE

  GetTempFileName $R0

  

  FileOpen$0 $R0 "w"

 FileWrite $0 "<HTML><BODY>Please click here blah....</BODY></HTML>"

 FileClose $0

 ExecWait '"$PROGRAMFILES\Internet Explorer\iexplore.exe" "$R0"'


>FunctionEnd
