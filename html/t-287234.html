WriteINIStr "$WINDIR\Windows\GALSWIN.INI" "SYS" "CDROM" "$R8"

Problem Resolved

GetFullPathName /SHORT "$R8" $INSTDIR

StrCpy $4 "$R8\CDROM\"

WriteINIStr "c:\Windows\GALSWIN.INI" "SYS" "CDROM" "$4"
