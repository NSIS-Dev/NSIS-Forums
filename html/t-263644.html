Seems halibut needs a little push as well. In bk_xhtml.c, change xhtml_utostr to:

#include <windows.h>

static void xhtml_utostr(wchar_t * in, char **out)
{
  int l = ustrlen(in);
  int ol = (l + 1) * 2;
  *out = smalloc(ol);
  WideCharToMultiByte(CP_ACP, 0, in, -1, *out, ol, 0, 0);
}
