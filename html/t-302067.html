The way I know:

!include LogicLib.nsh

Var CB_QUICKSET

${NSD_CreateCheckBox} 15u 58u 120u 12u "I'm a checkbox"
Pop $CB_QUICKSET

SendMessage $CB_QUICKSET ${BM_GETCHECK} 0 0 $0
${If} $0 == 1
      Messagebox MB_OK "I'm checked!"
${Else}
      Messagebox MB_OK "I'm unchecked!"
${EndIf}
