To fix this issue:

!include "WordFunc.nsh"
!insertmacro WordFind

!macro WordFindFixedCall STR DEL COL RET
${WordFind} ${STR} ${DEL} ${COL} ${RET}
StrCmp "${STR}" "${RET}" 0 +2
StrCpy ${RET} ""
!macroend
!define WordFindFixed `!insertmacro WordFindFixedCall`
