!define MUI_PAGE_CUSTOMFUNCTION_SHOW dir_show
!insertmacro MUI_PAGE_DIRECTORY

Function dir_show
  GetDlgItem $0 $HWNDPARENT 3
  EnableWindow $0 0
FunctionEnd
