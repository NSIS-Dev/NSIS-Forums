!define MUI_LANGDLL_ALWAYSSHOW ;you might not want this in the final installer, but use it during testing
!include MUI.nsh
!insertmacro MUI_RESERVEFILE_LANGDLL

!insertmacro MUI_LANGDLL_DISPLAY
