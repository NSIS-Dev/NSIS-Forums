MSG
24th May 2012, 12:58
ClearErrors
FileOpen $0 $INSTDIR\file.ini r
IfErrors done

loop:
FileRead $0 $1
IfErrors close
${If} $1 = [section]
loop2:
FileRead $0 $1
StrCpy $2 $1 1
${If} $2 == [
goto close
${EndIf}
StrCpy $2 $1 4
${If} $2 == that
StrCpy $2 $1 "" 5
DetailPrint $2
${EndIf}
goto loop2
${EndIf}
goto loop

close:
FileClose $0
done:


Note that this assumes that there's only one [section] block.
