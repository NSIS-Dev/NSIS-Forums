Format is ok, but sometimes it needs to be in quotes. You may also add buttons colors. Below my example:

##Interface setup
!define UMUI_SKIN "skinname"
!define UMUI_HEADERBGIMAGE
!define UMUI_TEXT_COLOR "FFFFFF"
!define MUI_BGCOLOR "2860A8"
!define UMUI_HEADERTEXT_COLOR "FFFFFF"
#Text and background colors for all Input,List,Treeview (except for licencepage and instfile page)
!define UMUI_TEXT_INPUTCOLOR 000000
!define UMUI_BGINPUTCOLOR "FFFFFF"
!define MUI_LICENSEPAGE_BGCOLOR "FFFFFF"
!define MUI_INSTFILESPAGE_COLORS "${UMUI_TEXT_INPUTCOLOR} ${UMUI_BGINPUTCOLOR}"
#Use skinned buttons
!define UMUI_DISABLED_BUTTON_TEXT_COLOR 666666
!define UMUI_SELECTED_BUTTON_TEXT_COLOR 000066
!define UMUI_BUTTON_TEXT_COLOR 000000
