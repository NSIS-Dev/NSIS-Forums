Function pre
nsDialogs::Create 1018
Pop $dialog
${NSD_CreateCheckbox} 0 0 50% 6% "I do not quit"
Pop $hwnd
${NSD_OnClick} $hwnd EnDisableButton

nsDialogs::Show
FunctionEnd

Function EnDisableButton
Pop $hwnd
${NSD_GetState} $hwnd $0
${If} $0 == 1
Quit ;-----I do not quit
