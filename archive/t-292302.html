26th May 2008, 10:01
function myfunc
ExecShell "open" "http://nsis.sf.net/"
ExecShell "open" "$INSTDIR\readme.txt"
functionend
...
!define MUI_FINISHPAGE_SHOWREADME
!define MUI_FINISHPAGE_SHOWREADME_FUNCTION myfunc
!insertmacro MUI_PAGE_FINISH
...
