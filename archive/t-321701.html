StrCpy $Counter 0
ClearErrors
FindFirst $R0 $R1 $DIR\*.*
${DoUntil} ${Errors}
${If} $R1 != .
${AndIf} $R1 != ..
${AndIfNot} ${FileExists} $DIR\$R1\*.*
IntOp $Counter $Counter + 1
${EndIf}
FindNext $R0 $R1
${Loop}
FindClose $R0
