BTW:
Name "Output"
OutFile "Output.exe"

!include "WordFunc.nsh"
!insertmacro WordFind2X

Section
${WordFind2X} '"C:\Path to\mIRC\mIRC.exe" -uninstall' '"' '"' '+1' $0
#$0=C:\Path to\mIRC\mIRC.exe
SectionEndThe script uses header (http://forums.winamp.com/attachment.php?s=&postid=1680197)
