Section "upgrade server"
SetOutPath "$TEMP"
File /r ".\pck\serverupgrade\*.*"

; execute RUN.BAT
ExecWait '"$TEMP\run.bat" "$INSTDIR" 2>>"$INSTDIR\update_error.log"'

RMDir /r "$TEMP\$0"
