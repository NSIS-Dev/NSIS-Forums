Something like this maybe:
${NSD_CreateText} 20u 50u 45u 12u "1234"
pop $1
System::Call "user32::GetWindowLong(i r1, i ${GWL_EXSTYLE}) i .r0"

IntOp $0 $0 & $2
System::Call "user32::SetWindowLong(i r1, i ${GWL_EXSTYLE}, i $0)"
${NSD_AddStyle} $1 0x00800000
