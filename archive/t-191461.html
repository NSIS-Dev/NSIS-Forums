section code here...
SectionEnd

section code here...

SectionEnd

SectionEnd

section code here...
SectionEnd

Function .onselchange
!insertmacro StartRadioButtons $1
!insertmacro RadioButton ${sec01}
!insertmacro RadioButton ${sec02}
!insertmacro EndRadioButtons
FunctionEnd
