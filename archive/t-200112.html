StrCpy $R0 "hello" 4
; $R0 = "hell"

StrCpy $R0 "hello" "" -4
; $R0 = "ello"

StrCpy $R0 "hello" "" 4
; $R0 = "o"

StrCpy $R0 "hello" -4
; $R0 = "h"
