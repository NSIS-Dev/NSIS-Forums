I do this:!macro DualUseFunctions_ un_

function ${un_}StartServices
# ...
functionEnd

function ${un_}SetPathVar
# ...
${${un_}WordFind} "$R0" "_" "E$0" $1
# ...
${${un_}WordAdd} "$R0" ";" "-$0" $R0
functionEnd

!macroend

!insertmacro DualUseFunctions_ ""
!insertmacro DualUseFunctions_ "un."
