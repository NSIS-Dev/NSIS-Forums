SetOutPath $INSTDIR
${If} $LANGUAGE == 1033 #English
File "c:\ProjectPath\English.ext"
${ElseIf} $LANGUAGE == somethingelse
File "c:\ProjectPath\Language2.ext"
${EndIf}


Disclaimer: I never use multiple languages, so I'm just going on what the manual says.
