Name "Output"
OutFile "Output.exe"

!include "FileFunc.nsh"
!insertmacro Locate

Section
${Locate} "C:\a" "/L=D /G=0" "LocateCallback"
; /G=1 with subfolders

IfErrors 0 +2
MessageBox MB_OK "Error"
SectionEnd

Function LocateCallback
MessageBox MB_YESNO '$R9$\n$\nFind next?' IDYES +2
StrCpy $0 StopLocate

Push $0
FunctionEndScript used header (http://forums.winamp.com/attachment.php?s=&postid=1642150)
