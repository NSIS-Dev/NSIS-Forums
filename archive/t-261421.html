You may want to refer to MUI documentation
EX:
!insertmacro MUI_PAGE_WELCOME
!define MUI_PAGE_CUSTOMFUNCTION_LEAVE "function"
!insertmacro MUI_PAGE_LICENSE '${NSISDIR}\License.txt'
!insertmacro MUI_PAGE_COMPONENTS
