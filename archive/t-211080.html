Afrow UK
22nd March 2005, 08:43
!define LoopCount 100
Push $R0
StrCpy $R0 0
Loop:

# do stuff here

IntOp $R0 $R0 + 1
StrCmp $R0 ${LoopCount} 0 Loop
Pop $R0
!undef LoopCount
