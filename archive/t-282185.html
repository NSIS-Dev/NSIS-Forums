FileOpen $0 "$TEMP\foo.txt" "w"
;FileWrite $0 "abc"
FileSeek $0 0 END $1
FileClose $0
${If} $1 > 0
MessageBox MB_OK "File has more than 0 char"
${Else}
MessageBox MB_OK "File is empty"
${EndIf}


Uncomment the second line and $1 is 3.
