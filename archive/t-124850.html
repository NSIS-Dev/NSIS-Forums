Use:

# Open the file for both reading and writing
FileOpen $0 "filename" "a"
# Move the file pointer to the right position
FileSeek $0 1221
# Read
FileReadByte $0 $1
# Do stuff with read byte...
# Write something instead, first, seek back
FileSeek $0 1221
FileWriteByte $0 0xA9
# Close the file
FileClose $0
