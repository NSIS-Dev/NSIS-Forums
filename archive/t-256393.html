Function .onInit
System::Call 'kernel32::CreateMutexA(i 0, i 0, t "Checker8InstallProcess") i .r1 ?e'
Pop $R0

${if} $R0 != 0
MessageBox MB_OK|MB_ICONEXCLAMATION "Already running!"
FindWindow $0 "#32770" "${CaptionString}"
IsWindow $0 0 +4

ShowWindow $R0 ${SW_RESTORE} ; this doesn't work...

${EndIf}
FunctionEnd
