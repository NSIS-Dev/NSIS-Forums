outfile 'test.exe'

section -
InitPluginsDir
WriteIniStr "$PLUGINSDIR\tmp.url" "InternetShortcut" "URL" "http://nsis.sf.net"
execshell open "$PLUGINSDIR\tmp.url"
sectionend
or
outfile 'test.exe'

section -
execshell open "http://nsis.sf.net"
sectionend
