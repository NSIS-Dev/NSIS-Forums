Use the logic library to be able to use if-else-end logic.

!include LogicLib.nsh

${If} ${FileExists} $WINDIR\notepad.exe
MessageBox MB_OK "notepad found"
${Else}
MessageBox MB_OK "notepad not found"
${EndIf}
