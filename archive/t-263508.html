var myvar

Function .onUserAbort
MessageBox MB_YESNO "Abort install?" IDYES NoCancelAbort
Abort ; causes installer to not quit.
NoCancelAbort:
StrCpy $myvar 'aborted'
FunctionEnd

Function .onGUIEnd
StrCmp $myvar 'aborted' end
Exec 'other_installer'
end:
FunctionEnd
