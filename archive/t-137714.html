!macro IfFileInUse FILE GOTONOTLOCKED GOTOLOCKED

ClearErrors

Push $R0

FileOpen $R0 "${FILE}" "a"
FileClose $R0

Pop $R0

IfErrors "${GOTONOTLOCKED}" "${GOTOLOCKED}"

!macroend
Usage:

!insertmacro IfFileInUse "c:\bla.txt" notlocked locked
