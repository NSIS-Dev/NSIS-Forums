Something like:

!include WordFunc.nsh
Function .onInit
${GetParameters} $1
${GetOptions} $1 "-CopiedExe=" $2
${If} $1 != 1
InitPluginsDir
CopyFiles /SILENT $EXEPATH $PLUGINSDIR
Exec "$PLUGINSDIR\$EXEFILE -CopiedExe=1"
Quit
${EndIf}
FunctionEnd
