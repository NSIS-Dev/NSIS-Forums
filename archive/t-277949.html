!define MUI_FINISHPAGE_TITLE_3LINES
!define MUI_WELCOMEPAGE_TITLE_3LINES

!insertmacro MUI_PAGE_*

!define MUI_FINISHPAGE_TITLE_3LINES
!define MUI_WELCOMEPAGE_TITLE_3LINES

!insertmacro MUI_UNPAGE_*
