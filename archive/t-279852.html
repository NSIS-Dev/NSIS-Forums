Easily, add FILE_FLAG_BACKUP_SEMANTICS flag to CreateFile function....
// Get the handle of the file object.
hFile = CreateFile(
argv[1],
GENERIC_READ,
FILE_SHARE_READ,
NULL,
OPEN_EXISTING,
FILE_ATTRIBUTE_NORMAL | FILE_FLAG_BACKUP_SEMANTICS,
NULL);
...
