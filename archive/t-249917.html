29th June 2006, 08:36
Name "Output"
OutFile "Output.exe"

Section
System::Call "kernel32::GetCurrentDirectory(i ${NSIS_MAX_STRLEN}, t .r0)"
MessageBox MB_OK "$0"
SectionEnd
