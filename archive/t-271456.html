You can call GetDriveType using the System plug-in and compare the result to DRIVE_REMOTE.!include LogicLib.nsh
#...
!define DRIVE_REMOTE 4
System::Call 'kernel32::GetDriveType(t "J:\") i.r0'
${If} $0 == ${DRIVE_REMOTE}
# bad installation directory
${EndIf}
