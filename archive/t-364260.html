You can get the printed text from a command with nsExec::ExecToStack (and skip batch file):nsExec::ExecToStack '"perl.exe" -Version'
Pop $0 # return value/error/timeout
Pop $1 # printed text
StrCpy $1 "$1" "" -18 # only keep last 18 characters from the returned string
${If} $1 == "perl not installed"
goto install
${EndIf}
