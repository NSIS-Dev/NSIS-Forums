${FileJoin} "$INSTDIR\cstrike\autoexec.cfg" "$INSTDIR\Infos\exec.cfg" "$INSTDIR\cstrike\autoexec.cfg"with IfFileExists "$INSTDIR\cstrike\autoexec.cfg" +2
