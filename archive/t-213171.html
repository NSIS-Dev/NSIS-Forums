Name "Output"
OutFile "Output.exe"

!include "TextFunc.nsh"
!insertmacro LineFind

Section
${LineFind} "C:\input.txt" "C:\output.txt" "1:7 -3:-1" "LineFindCallback"

IfErrors 0 +2
MessageBox MB_OK "Error"
SectionEnd

Function LineFindCallback
StrCpy $0 SkipWrite

Push $0
FunctionEnd
