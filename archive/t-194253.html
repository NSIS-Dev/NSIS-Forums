This can be done dynamically like so...

Section "My Section" MySec
...
SectionEnd

!include Sections.nsh

Function Blah

;Enable section
!insertmacro SelectSection "${MySec}"

;Disable section
!insertmacro UnselectSection "${MySec}"

FunctionEnd
