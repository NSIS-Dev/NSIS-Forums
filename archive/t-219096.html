Name "Output"
OutFile "Output.exe"

!include "FileFunc.nsh"
!insertmacro GetParameters
!insertmacro GetOptions

Section
#CMDLINE:
#Output.exe /TEST="C:\Program Files" /Something else=0

${GetParameters} $0
${GetOptions} '$0' '/TEST=' $1

#$1=C:\Program Files
SectionEndScript use header (http://forums.winamp.com/attachment.php?s=&postid=1680197)
