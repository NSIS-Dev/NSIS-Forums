As I said,


!define MUI_WELCOMEPAGE_TEXT "My installer text"
!insertmacro MUI_PAGE_WELCOME

!define MUI_WELCOMEPAGE_TEXT "My uninstaller text"
!insertmacro MUI_UNPAGE_WELCOME
