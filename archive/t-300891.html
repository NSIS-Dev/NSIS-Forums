Ok here is the code for searching and deleting.

------------------------------------------------------------
outFile "Testme.exe"

!include "logiclib.nsh"
!include "xml.nsh"

Section

${xml::LoadFile} "c:\Test.xml" $0

${Do}
${xml::FindNextElement} "Child" $0 $2

${If} $2 == -1
${xml::FindCloseElement}
${ExitDo}
${EndIf}

${xml::FirstChildElement} "Name" $8 $9
${xml::GetText} $8 $9

${If} $8 == 'Item -2'

${xml::Parent} $0 $1
${xml::RemoveNode} $0
${xml::SaveFile} "" $0

${ExitDo}
${EndIf}
${Loop}

SectionEnd
