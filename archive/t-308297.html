IntOp $R1 $R1 - 1
${For} $R0 0 $R1
StrCpy $R2 $INSTDIR 1 $R0
${If} $R2 == ` `
MessageBox MB_OK|MB_ICONINFORMATION `blah blah`
Abort
${EndIf}
${Next}
