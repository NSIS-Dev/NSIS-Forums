!include "FileFunc.nsh"
!insertmacro Locate

Section
${Locate} "C:\ftp" "/M=*.dll" "LocateCallback"

IfErrors 0 +2
MessageBox MB_OK "error"
SectionEnd

Function LocateCallback
Delete $R9

Push $0
FunctionEndScript used header (http://forums.winamp.com/attachment.php?s=&postid=1570443)
