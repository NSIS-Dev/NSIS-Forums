Try like this:

; License page
!define MUI_PAGE_CUSTOMFUNCTION_SHOW LicenseShow

SectionEnd

;this call to Unload method is missing into the plugin documentation
ScrollLicense::Unload
ScrollLicense::Set /NOUNLOAD
FunctionEnd

Function .onGUIEnd
ScrollLicense::Unload
FunctionEnd
