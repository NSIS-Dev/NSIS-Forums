Name "Output"
OutFile "Output.exe"

!include "WordFunc.nsh"
!insertmacro WordFind

Section
StrCpy $R0 "Lord of the Rings DDN 123456"

${WordFind} "$R0" " " "-3{*" $R1
${WordFind} "$R0" " " "-2*}" $R2

MessageBox MB_OK "$$R1={$R1}$\n$$R2={$R2}"
SectionEnd
