CopyFiles?
CreateDirectory "$INSTDIR\Backup"

SetOutPath $INSTDIR

IfFileExists "$INSTDIR\file.ext" 0 +2
CopyFiles "$INSTDIR\file.ext" "$INSTDIR\Backup\file.ext"

File "file.ext"
File "..."

RMDir "$INSTDIR\Backup" ; remove backup dir if empty
