Compile Time Commands
http://nsis.sourceforge.net/Docs/Chapter5.html

variant1.nsi:
!define VARIANT "1"
!include main.nsi

variant2.nsi:
!define VARIANT "2"
!include main.nsi

main.nsi:
!if "${VARIANT}" == "1"
Name "Installer Variant One"
!else if "${VARIANT}" == "2"
Name "Installer Variant Two
!endif
