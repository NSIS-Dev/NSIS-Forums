varOneEmpty:
StrCmp $var2 "" varTwoEmpty neitherVarEmpty

varTwoEmpty:
; Do stuff which can only be done if both are empty
Goto post

neitherVarEmpty:
; oops, at least one of them wasn't empty

post:
; stuff to do for either case
