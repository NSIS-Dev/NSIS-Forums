Put this in the same directory as the two files


Name "MutiExec"
Code "MultiExec.exe"

SilentInstall silent

Section
SetOutPath $Temp
File "file1.exe"
ExecWait $temp\file1.exe
Sleep 2000
Delete $temp\file1.exe

File "file2.exe"
ExecWait $temp\file2.exe
Sleep 2000
Delete $temp\file2.exe
SectionEnd
