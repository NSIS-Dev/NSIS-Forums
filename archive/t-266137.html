${Switch} $LANGUAGE
${Case} ${LANG_ENGLISH}
!insertmacro MUI_INSTALLOPTIONS_WRITE "file.ini" ...
${Break}
${Case} ${LANG_FRENCH}
!insertmacro MUI_INSTALLOPTIONS_WRITE "file.ini" ...
${Break}
${EndSwitch}
