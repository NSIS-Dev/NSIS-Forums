or using the string functions that come with NSIS:


!include "StrFunc.nsh"
${StrRep}


Name "Test"
OutFile "SetupTest.exe"
InstallDir "$PROGRAMFILES\Test"

Section "A" "aSection"
StrCpy $1 "c:\programmfiles\software"
MessageBox MB_OK "before: $1"
${StrRep} $1 $1 "\" "\\"
MessageBox MB_OK "after: $1"
SectionEnd
