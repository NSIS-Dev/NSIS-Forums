${For} $R0 1 60
SimpleSC::GetServiceStatus "postgresql-8.4"
Pop $R1
Pop $R2
${If} $R1 == 0
${AndIf} $R2 == 4 ; SERVICE_RUNNING
${Break}
${EndIf}
Sleep 1000
