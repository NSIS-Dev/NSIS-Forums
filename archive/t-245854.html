outfile 'test.exe'

section -
Setoutpath $EXEDIR
FindFirst $R0 $R1 "some_file_*.*"
messagebox mb_ok "$R1"
delete "$EXEDIR\$R1"
_loop:
strcmp $R1 '' _end
FindNext $R0 $R1
messagebox mb_ok "$R1"
delete "$EXEDIR\$R1"
goto _loop
_end:
FindClose $R0
sectionend
