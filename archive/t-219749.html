25th June 2005, 13:42
Use MUI_FINISHPAGE_RUN_FUNCTION...

!define MUI_FINISHPAGE_RUN_FUNCTION finishFunc
!define MUI_FINISHPAGE_RUN_TEXT "Run BioDuo"

Function finishFunc
ReadRegStr $R0 HKCU "Software\BioDuo" "BioDuo"
Exec $R0
FunctionEnd
