!define CM_REENUMERATE_SYNCHRONOUS 1
!define CM_REENUMERATE_RETRY_INSTALLATION 2
!define CM_REENUMERATE_ASYNCHRONOUS 4

System::Call 'CFGMGR32::CM_Reenumerate_DevNode(i 0, i ${CM_REENUMERATE_SYNCHRONOUS})'You'd probably need to replace the 0 with a real DEVNODE handle.
