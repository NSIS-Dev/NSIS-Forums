Easy and faster without parsing extended file, but if you want (required (http://nsis.sourceforge.net/wiki/Registry_plugin)):

Name "Output"
OutFile "Output.exe"

!include "TextFunc.nsh"
!insertmacro LineFind
!insertmacro TrimNewLines

!include "WordFunc.nsh"
!insertmacro WordFind

Section
${LineFind} "C:\input.txt" "/NUL" "1:-1" "LineFindCallback"

IfErrors 0 +2
MessageBox MB_OK "Error"
SectionEnd

Function LineFindCallback
${TrimNewLines} "$R9" $R9
${WordFind} "$R9" " " "+1{" $0
${WordFind} "$R9" '"' "+2" $1
${WordFind} "$R9" '"' "+4" $2
${WordFind} "$R9" '"' "+6" $3

registry::Write "$0\$1" "$2" "$3" "REG_SZ" .R0

Push 0
FunctionEnd
